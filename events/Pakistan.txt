country_event = { # muslim league accepts invite IF you support an infrastructure project
	id = pakistan.4
	title = pakistan.4.t
	desc = pakistan.4.d
	#picture = #todo

	fire_only_once = yes

	is_triggered_only = yes

	option = { # say no - forces you down streamline status quo path
		name = pakistan.4.a
		add_political_power = 5
		complete_national_focus = PAK_streamline_the_status_quo
	}

	option = { # say yes - gain 3 infrastructure in Sindh, lose 1% stability, get spirit giving consumer goods use +5% for 2 months
		name = pakistan.4.b

		add_stability = -0.01
		443 = {
			add_building_construction = {
				type = infrastructure
				level = 3
				instant_build = yes
			}
		}
		add_timed_idea = { idea = PAK_expensive_infrastructure_project days = 60 }
		set_country_flag = PAK_pakistan_peoples_party_invitation
		set_country_flag = PAK_awami_league_ready # move to awami league invite focus if this was chosen first
	}
}
